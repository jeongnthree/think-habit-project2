아래 코드를 복사하여 홈페이지 HTML에 붙여넣으세요:

<!-- Think-Habit 위젯 시작 --> <div id="think-habit-widget"></div> <script> // 위젯 설정 const widgetConfig = { groupId: 'church-2024', // 교회 고유 ID theme: 'light', // 테마 height: 600, // 높이 showLeaderboard: true, // 순위표 표시 showRecentActivity: true, // 활동 피드 표시 showJournalForm: true, // 일지 작성 폼 표시 language: 'ko' // 한국어 }; // 위젯 로드 (function() { // 실제 사용 시에는 Think-Habit 공식 CDN URL로 변경하세요 const script = document.createElement('script'); script.src = '/widget/index.js'; script.onload = function() { if (window.ThinkHabitWidget) { window.ThinkHabitWidget.init('think-habit-widget', widgetConfig); } }; document.body.appendChild(script); })(); </script> <!-- Think-Habit 위젯 끝 -->
